// check if is the DYOM current game script
// return @0 - true or false
:CHECK_FOR_DYOM
    if CHECK_FOR_DYOM_81()
    then
        return 1 true
    end
    
    if CHECK_FOR_DYOM_RED()
    then
        return 1 true
    end
            
    //print_formatted_now "DYOM not detected. Memory markers 0x%x 0x%x" time 2000 $9877 $9888 // DEBUG  
return 1 false


:CHECK_FOR_DYOM_81
    if and // DYOM 8.1 and other compatible
        $9877 == 0xA49960
        $9888 == 0xA5C2D2
    then
        return 1 true
    end
return 1 false


:CHECK_FOR_DYOM_RED
    if and // DYOM REd
        $9877 == 0xA4995B
        $9888 == 0xA5C2CD
    then
        return 1 true
    end
return 1 false
